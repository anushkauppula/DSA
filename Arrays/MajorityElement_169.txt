-- Optimized O(n), O(1)
class Solution {
    public int majorityElement(int[] nums) {
        int count = 0; int element = 0;

        for(int num : nums){
            if(count == 0){
                element = num;
            }

            count += (element == num) ? 1 : -1;
        }
        return element;
    }
}
