--Optimized O(log n*m), O(1)
class Solution {
    public boolean searchMatrix(int[][] matrix, int target) {
        int rows = matrix.length;
        int cols = matrix[0].length;
        int len = rows * cols;

        int low = 0; int high = len - 1;
        while(low <= high){
            int mid = (low + high)/2;

            int row = mid/cols;
            int col = mid%cols;
            if(matrix[row][col] == target){
                return true;
            }else{
                if(matrix[row][col] < target){
                    low = mid + 1;
                }else{
                    high = mid - 1;
                }
            }
        }

        return false;


    }
}
