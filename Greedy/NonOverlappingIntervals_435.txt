// O(n log n) + O(n), O(1)
class Solution {
    public int eraseOverlapIntervals(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[1] - b[1]); //TC - O(n log n)
        int count = 1; 
        int lastEndValue = intervals[0][1];

        for(int[] i : intervals){
            if(i[0] >= lastEndValue){
                count = count + 1;
                lastEndValue = i[1];
            }
        }
        
        return intervals.length - count;
    }
}
